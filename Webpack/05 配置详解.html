Webpack 配置详解

Re_Star 2020-01-06 webpack

# entry

含义：打包的入口文件

属性值

  1. string --> './src/index.js'
单入口
打包形成一个chunk,输出一个bundle文件
此时chunk的名称默认为main
  1. array ---> ['./src/add.js' , './src/index.js']
多入口
所有的入口文件最终只会形成一个chunk，输出出去只有一个bundle
只有在HMR功能中让html热更新生效
  1. object（文件名称：文件路径）
多入口
有几个入口文件就形成几个chunk，输出几个bundle文件
此时chunk的名称是key
特殊用法
{
    // 所有入口文件最终只会形成一个chunk,输出一个bundle
   	“index":['index.js','count.js'],
   	// 形成一个chunk,输出一个bundle
   	'add':'./src/add.js'
}

# output

基本配置

....
module.exports = {
    entry:'./src/index.js',
    output:{
        // 文件名称(指定名称+目录)
        filename:'js/[name].js',
        // 输出文件目录(将来所以输出资源的公共目录)
        path:resolve(__dirname,'build'),
        // 所有输出资源引用公共路径 --> 路径的前面
        publicPath:'/',
        // 非chunk的名称
        chunkFilename:'[name]_chunk.js',
        library:'[name]', // 整个库向外暴露的变量名
        libraryTarget:'window' // 变量名添加到哪一个上面 browser 添加到window
        libraryTarget:'global' // 变量名添加到哪一个上面 node 添加到global
        libraryTarget:'commonjs' // 以
    }	
}

# module

module:{
    rules:[
        //loader 配置
        {
            test:/\.css$/,
            // 多个loader用use
            use:['style-loader','css-loader']
        },
        {
            test:/\.js$/,
            // 排除/node_modules/下的js文件
            exclude: /node_modules/,
            // 只检查src的js文件
            include:resolve(__dirname,'src')
            // 优先执行
            // enforce:'pre',
            // 延后执行
            // enforce:'post',
            // 多个loader用loader
            loader:'eslint-loader',
            // 给loader传递参数
            options:{}
        },
        {
            // 以下配置只生效一个
            oneOf:[]
        }
    ]
}

# resolve

用于解析模块规则

module.exports = {
    // 解析模块的规则
    resolve:{
        // 配置解析模块路径别名
        // 可以简写引入模块的路径
        // 代码语义不明
        alias:{
            $css:resolve(__dirname,'src/css')
        }，
        // 配置省略文件路径的后缀名
        extensions:['js','json','css','.jsx'],
    	// 告诉webpack解析模块去哪个目录找
    	modules:[resolve(__dirname,'../../node_modules')'node_modules']
    }
}

# devServer

module.exports = {
    devServer:{
        // 运行代码的目录
        contentBase:resolve(__dirname,'build'),
        // 监视contentBase目录下的所有文件，一旦文件变化就会reload
        watchContentBase:true,
        // 监视配置
        watchOptions:{
        	// 忽略文件
            ignored:/node_modules/            
        },
        // 启动gzip压缩
        compress:true,
        // 端口号
        port:5000,
        // 域名
        host:'localhost'
        // 自动打开浏览器
        open:true,
        // 开启HMR功能
        hot:true,
        // 服务器日志相关配置信息
        clientLogLevel:'none'
        // 除了一些基本的启动信息以外，其他内容都不要显示	
        quiet:true,
        // 如果出现错误，不要全屏提示
        overlay:false
        // 服务器代理 --> 解决开发环境跨域问题
        proxy:{
        	// 一旦devServer(5000)服务器接收到/api/xxx的请求，就会把请求转发到端口号3000
        	'/api':{
        		target:'http://localhost:3000',
        		// 发送请求时，请求路径重写 将 '^/api/xxx ---> /xxx'
        		pathRewrite:{
        			'^/api':''
    			}
    		}
    	}
    }
}

# optimization

module.exports = {
    
    optimization:{
        splitChunks :{
            chunks:'all',
            // 以下都是默认值
            minSize:30*1024, // 分割的chunk最小为30kb
            maxSize:0, // 最大没有限制
            minChunks:1 // 要提取的chunk最少次数为1次
            maxAsyncRequests:5, // 按需加载时并行加载的文件最大数量
            minInitialRequest:3, // 入口js文件最大并行请求数量
            automaticeNameDelimiter:'~' // 名称连接符
            name:true, // 可以使用命名规则
            // 分割chunk的组
            cacheGroups:{
            	// node_modules文件会被打包到vendors组的chunk中 --> vendors~xxx.js
            	// 还要满足上面的公共规则
            	vendors:{
            		test:/[\\/]node_modules[\\/]/,
            		// 打包的优先级
            		priority:-10
        		},
        		default:{
                    // 要提取的chunk最少次数为2次
            		minChunks:2,
                    // 打包的优先级
            		priority:-20,
                    // 如果当前要打包的模块，和之前已经被提取的模块是同一个，就会复用，而不是重新打包模块
                    reuseExistingChunk:true
        		}
        	}
        },
        runtimeChunk:{
            // 将当前模块记录的引入其他模块哈希值打包成一个单独文件
            name:entrypoint=>`runtime-${entrypoint.name}`
        },
         minimizer:[
             // 配置js和Css的压缩方案
             new TerserWebpackPlugin({
                 // 开启缓存
                 cache:true,
                 // 开启多进程打包
                 parallel:true,
                 // 开启sourceMap
                 sourceMap:true
             })
         ]       
    }
    
}
