Webpack 开发配置

Re_Star 2019-12-24 webpack

# CSS

相关包：style-loader,css-loader

使用，在loade中使用

module.exports = {
   //  ....
    module:{
    	rules:[
            {
                test:/\.css$/,
                use:['style-loader','css-loader']
            },
            {
                test:/\.less$/,
                use:['style-loader','css-loader','less-loader']
            }
        ]
	}
}

# JavaScript

webpack本身就支持js和json

但是为了考虑js代码的兼容性，我们会使用babel来做预处理

相关包：babel-loader, @babel/core

module.exports = {
   //  ....
    module:{
    	rules:[
            {
            	test:/\.js$/,
                // 让babel跳过/node_modules/文件夹下的编译
                exclude:/node_modules/,
                loader:'babel-loader'
            }
        ]
	}
}

# HTML

相关包：html-webpack-plugin

const HtmlWebpackPlugin = require('html-webpack-plugin')

module.exports = {
   //  ....
  	plugins:[
        new HtmlWebpackPlugin({
            template:'index.html'
        })
    ]
}

# Image

相关包：url-loader,html-loader(用于处理html文件中引入的图片资源)

module.exports = {
   //  ....
    module:{
    	rules:[
           {
               test:/\.(jpg|png|gif|jpeg)$/,
               loader:'url-loader',
               options:{
                   limit:8*1024,
                   esModule:false
               }
           },
           {
               test:/\.html$/,
               loader:'html-loader'
           }
        ]
	}
}

# others

例如字体文件，图标文件等

相关包：file-loader

module.exports = {
   //  ....
    module:{
    	rules:[
           {
               exclude:/\.(html|css|js|png|gif|jpg|jpeg)$/,
               loader:'file-loader'
           }
        ]
	}
}

# 总和

下面是一个最简单的webpack运行配置

const { resolve } = require('path');
const HtmlWebpackPlugin = require('html-webpack-plugin')

module.exports = {
    // 入口文件
    entry:'./src/main.js'
    // 输出文件
    output:{
    	filename:'js/built.js',
    	path:resolve(__dirname,'build')
	},
    // loader
    module:{
        rules:[
            {
                test:/\.css$/,
                use:['style-loader','css-loader']
            },
            {
            	test:/\.js$/,
                // 让babel跳过/node_modules/文件夹下的编译
                exclude:/node_modules/,
                loader:'babel-loader'
            },
            {
               test:/\.(jpg|png|gif|jpeg)$/,
               loader:'url-loader',
               options:{
                   limit:8*1024,
                   esModule:false
               }
            },
           	{
               test:/\.html$/,
               loader:'html-loader'
           	},
            {
               exclude:/\.(html|css|js|png|gif|jpg|jpeg)$/,
               loader:'file-loader'
           	}
        ]
    },
    // 插件
    plugins:[
           new HtmlWebpackPlugin() 
    ],
    // 运行模式,development
    mode:'development',
        
}
